Обновление и создание товара в заказе:
  
  запрос от web или core:
  {
    name: order_product.update,

    data: {
      id:         986,
      product_id: 24,
      sum:        500,
      quantity:   10
    },

    created: 2011-07-22 10:01:06
  }

  запрос от web:
  {
    name: order_product.create,

    data: {
      order_id:   986,
      product_id: 24,
      sum:        500,
      quantity:   10
    },

    created: 2011-07-22 10:01:06
  }



  ответ в случае успеха:
  {
    success: true        # успешность выполнения запроса
    data: {
      order_product: { id: 100 } # порядок, создан товар в заказе с ид = 100
    }
  }



  ответ в случае ошибки:
  {
    success: false
    error: {
      code: 45
      name: 'Невозможно добавить товар в заказ'
    }
  }
